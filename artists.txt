  function escapeHtml(str){
    return String(str)
      .replaceAll("&", "&amp;")
      .replaceAll("<", "&lt;")
      .replaceAll(">", "&gt;")
      .replaceAll('"', "&quot;")
      .replaceAll("'", "&#39;");
  }

  function renderArtistCard(a){
    const name = escapeHtml(a.name);
    const img = escapeHtml(a.img || PLACEHOLDER_IMG);
    const x = (a.x || "").trim();

    const linkHtml = x
      ? `<a class="cfArtistBtn" href="${escapeHtml(x)}" target="_blank" rel="noopener noreferrer">Xを見る ↗</a>`
      : `<div class="cfArtistBtn is-disabled">X（準備中）</div>`;

    return `
      <article class="cfArtistCard" id="${escapeHtml(a.id)}">
        <div class="cfArtistThumb">
          <img src="${img}" alt="${name}" loading="lazy">
        </div>
        <div class="cfArtistBody">
          <h3 class="cfArtistName">${name}</h3>
          ${linkHtml}
        </div>
      </article>
